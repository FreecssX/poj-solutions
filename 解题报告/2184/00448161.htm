int  TotalMax, nNum, Si[100], Fi[100];

int GetMax(int TotalSi, int TotalFi, int Order){
	if (TotalSi >= 0 && TotalFi >=0){
		if (TotalSi + TotalFi > TotalMax) TotalMax = TotalSi + TotalFi;
		return TotalSi + TotalFi;
	}
	if (Order >= nNum || TotalSi + TotalFi <= TotalMax)
		return 0;
	int i = 0, j = 0;
	i = GetMax(TotalSi, TotalFi, Order + 1);
	j = GetMax(TotalSi + Si[Order], TotalFi + Fi[Order], Order + 1);
	return i > j ? i : j;
}

			if (Si[nNum] + Fi[nNum] >= 0){
